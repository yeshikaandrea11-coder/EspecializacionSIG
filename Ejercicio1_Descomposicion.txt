n=int
divisor=int
n = int(input("Ingrese un numero mayor que 1: "))
if n <= 1:
    print("Error: el numero debe ser mayor que 1")
else:
    print("Factores primos de", n, ":")
    divisor = 2 
    while n > 1:
        if n % divisor == 0:
            print(divisor)      
            n = n // divisor   
        else:
            divisor = divisor + 1
