r = int(input("Ingrese el primer numero (r): "))
n = int(input("Ingrese el segundo numero (n): "))
def factorial(x):
    f = 1
    i = 1
    while i <= x:
        f = f * i
        i = i + 1
    return f
if r < 0 or n < 0:
    print("Error: Los nÃºmeros deben ser numeros positivos")
elif r > n:
    print("Error: r no puede ser mayor que n")
else:
    comb = factorial(n) // (factorial(r) * factorial(n - r))

    print("La combinacion C(" + str(n) + "," + str(r) + ") es:", comb)
